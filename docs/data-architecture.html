Primary Database: queue.db (72 MB)
projects ~40K rows - all active queue entries
▼
CREATE TABLE projects (
  id INTEGER PRIMARY KEY,
  queue_id TEXT NOT NULL,
  region TEXT NOT NULL, -- NYISO, ERCOT, MISO, PJM, CAISO, SPP, ISO-NE
  name TEXT,
  developer TEXT,
  capacity_mw REAL,
  type TEXT, -- Solar, Wind, Storage, Gas, etc.
  status TEXT,
  state TEXT,
  county TEXT,
  poi TEXT, -- Point of Interconnection
  queue_date TEXT,
  cod TEXT, -- Commercial Operation Date
  source TEXT, -- nyiso, ercot, miso_api, pjm_direct, lbl
  raw_data TEXT, -- JSON blob
  row_hash TEXT, -- For change detection
  UNIQUE(queue_id, region, source)
);
